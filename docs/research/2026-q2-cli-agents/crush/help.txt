
  A glamorous, terminal-first AI assistant for software development and adjacent tasks                                  
         
  USAGE  
         
    crush [command] [--flags]                                                   
            
  EXAMPLES  
            
    # Run in interactive mode                                                   
    crush                                                                       
                                                                                
    # Run non-interactively                                                     
    crush run "Guess my 5 favorite Pokémon"                                     
                                                                                
    # Run a non-interactively with pipes and redirection                        
    cat README.md | crush run "make this more glamorous" > GLAMOROUS_README.md  
                                                                                
    # Run with debug logging in a specific directory                            
    crush --debug --cwd /path/to/project                                        
                                                                                
    # Run in yolo mode (auto-accept all permissions; use with care)             
    crush --yolo                                                                
                                                                                
    # Run with custom data directory                                            
    crush --data-dir /path/to/custom/.crush                                     
                                                                                
    # Continue a previous session                                               
    crush --session {session-id}                                                
                                                                                
    # Continue the most recent session                                          
    crush --continue                                                            
            
  COMMANDS  
            
    completion [command]                      Generate the autocompletion script for the specified shell
    dirs                                      Show config and data directories
    help [command]                            Help about any command
    login [platform] [--flags]                Login Crush to a platform
    logout [platform] [--flags]               Logout Crush from a platform
    logs [--flags]                            View crush logs
    models                                    List all available models from known providers
    projects [--flags]                        List project directories
    run [prompt...] [--flags]                 Run a single non-interactive prompt
    server [--flags]                          Start the Crush server
    session [command]                         Manage sessions
    stats                                     Show usage statistics
    update-providers [path-or-url] [--flags]  Update providers
         
  FLAGS  
         
    -C --continue                             Continue the most recent session
    -c --cwd                                  Current working directory
    -D --data-dir                             Custom crush data directory
    -d --debug                                Debug
    -h --help                                 Help
    -H --host                                 Connect to a specific crush server host (for advanced users) (unix:///var/folders/3b/zb2ty38s41jbqm1xcylmbrzr0000gn/T/crush-501.sock)
    -s --session                              Continue a previous session by ID
    -v --version                              Version for crush
    -y --yolo                                 Automatically accept all permissions (dangerous mode)
