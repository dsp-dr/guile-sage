#+TITLE: RFC-001: Native Plugin Contract
#+DATE: 2026-01-28
#+STATUS: Draft
#+AUTHOR: dsp-dr

* Abstract

This RFC proposes a native Guile plugin contract for extending guile-sage with external tools and capabilities. The design prioritizes simplicity, Guile idioms, and zero external dependencies.

* Motivation

guile-sage needs extensibility for:
- External service integrations (Cloudflare, Atlassian, databases)
- Custom tool sets for specific workflows
- Community contributions

Rather than adopt MCP (see ADR-001), we define a lightweight native contract.

* Design Goals

1. *Simple*: Minimal boilerplate, obvious semantics
2. *Guile-native*: Use modules, not JSON-RPC
3. *Safe by default*: Plugins declare required permissions
4. *Discoverable*: Runtime introspection of loaded plugins
5. *Hot-reloadable*: Support development workflow

* Plugin Contract

** Module Structure

Plugins are Guile modules under ~(sage plugins <name>)~:

#+begin_src scheme
;; src/sage/plugins/cloudflare.scm
(define-module (sage plugins cloudflare)
  #:use-module (sage plugin-api)
  #:export (plugin-manifest))

(define plugin-manifest
  `((name . "cloudflare")
    (version . "0.1.0")
    (description . "Cloudflare Workers and KV integration")
    (author . "dsp-dr")
    (tools . ,(list tool-cf-deploy
                    tool-cf-status
                    tool-cf-kv-get
                    tool-cf-kv-put))
    (init . ,plugin-init)
    (cleanup . ,plugin-cleanup)
    (permissions . (network shell))))
#+end_src

** Tool Definition

Tools follow the existing sage tool schema:

#+begin_src scheme
(define tool-cf-deploy
  `((name . "cf_deploy")
    (description . "Deploy a Cloudflare Worker")
    (parameters . ((type . "object")
                   (properties . ((script . ((type . "string")
                                             (description . "Worker script path")))
                                  (name . ((type . "string")
                                           (description . "Worker name")))))
                   (required . #("script" "name"))))
    (handler . ,cf-deploy-handler)
    (safe . #f)))  ; requires permission

(define (cf-deploy-handler args)
  (let ((script (assoc-ref args "script"))
        (name (assoc-ref args "name")))
    ;; Implementation
    (format #f "Deployed ~a from ~a" name script)))
#+end_src

** Lifecycle Hooks

#+begin_src scheme
;; Called when plugin is loaded
(define (plugin-init)
  ;; Initialize API client, validate credentials, etc.
  (let ((api-key (config-get "CLOUDFLARE_API_KEY")))
    (unless api-key
      (error "CLOUDFLARE_API_KEY not configured"))
    (set! *cf-client* (make-cf-client api-key))
    #t))

;; Called when plugin is unloaded
(define (plugin-cleanup)
  (when *cf-client*
    (cf-client-close *cf-client*)
    (set! *cf-client* #f))
  #t)
#+end_src

** Permissions Model

Plugins declare required permissions:

| Permission | Description |
|------------+-------------|
| ~network~ | HTTP/TCP connections |
| ~shell~ | Shell command execution |
| ~filesystem~ | File operations outside workspace |
| ~eval~ | Dynamic code evaluation |

#+begin_src scheme
;; In manifest
(permissions . (network))

;; Runtime check
(define (cf-deploy-handler args)
  (require-permission 'network)
  ;; ... implementation
  )
#+end_src

* Plugin API

** Loading Plugins

#+begin_src scheme
;; Load a plugin by name
(plugin-load 'cloudflare)

;; Load from path
(plugin-load-file "/path/to/plugin.scm")

#+end_src

** Unloading Plugins

#+begin_src scheme
;; Unload plugin (calls cleanup)
(plugin-unload 'cloudflare)

;; Reload plugin (for development)
(plugin-reload 'cloudflare)
#+end_src

** Tool Registration

When a plugin loads, its tools are registered with the main tool system:

#+begin_src scheme
;; Automatic on plugin-load
(plugin-load 'cloudflare)
;; Now (get-tool "cf_deploy") works

;; Tools are namespaced internally
;; plugin:cloudflare:cf_deploy
;; but accessible as just "cf_deploy" if unique
#+end_src

* Example: Minimal Plugin

#+begin_src scheme
;;; plugins/hello.scm - Minimal plugin example
(define-module (sage plugins hello)
  #:use-module (sage plugin-api)
  #:export (plugin-manifest))

(define (hello-handler args)

               (safe . #t))))

* Example: Service Integration

#+begin_src scheme
;;; plugins/atlassian.scm - Jira/Confluence integration
(define-module (sage plugins atlassian)
  #:use-module (sage plugin-api)
  #:use-module (sage util)  ; http-request, json-parse
  #:export (plugin-manifest))

(define *jira-client* #f)

(define (plugin-init)
  (let ((url (config-get "JIRA_URL"))
        (token (config-get "JIRA_API_TOKEN")))
    (unless (and url token)
      (error "JIRA_URL and JIRA_API_TOKEN required"))
    (set! *jira-client*
          `((url . ,url) (token . ,token)))
    #t))

(define (jira-create-issue args)
  (require-permission 'network)
  (let ((project (assoc-ref args "project"))
        (summary (assoc-ref args "summary"))
        (type (or (assoc-ref args "type") "Task")))
    (let ((response (http-request
                     (string-append (assoc-ref *jira-client* 'url)
                                    "/rest/api/2/issue")
                     #:method 'POST
                     #:headers `(("Authorization" .
                                  ,(string-append "Bearer "
                                                  (assoc-ref *jira-client* 'token)))
                                 ("Content-Type" . "application/json"))
                     #:body (json-encode
                             `((fields . ((project . ((key . ,project)))
                                          (summary . ,summary)
                                          (issuetype . ((name . ,type))))))))))
      (let ((key (assoc-ref (json-parse response) "key")))
        (format #f "Created issue: ~a" key)))))

(define plugin-manifest
  `((name . "atlassian")
    (version . "0.1.0")
    (description . "Jira and Confluence integration")
    (author . "dsp-dr")
    (tools . (((name . "jira_create_issue")
               (description . "Create a Jira issue")
               (parameters . ((type . "object")
                              (properties . ((project . ((type . "string")))
                                             (summary . ((type . "string")))
                                             (type . ((type . "string")))))
                              (required . #("project" "summary"))))
               (handler . ,jira-create-issue)
               (safe . #f))))
    (init . ,plugin-init)
    (permissions . (network))))
#+end_src

* Security Considerations (Red Team)

Self-modifying AI agents with plugin systems present unique attack surfaces.

** Threat Model

| Attacker | Goal | Vector |
|----------+------+--------|
| Malicious plugin | Exfiltrate data | Network permission abuse |
| Compromised LLM | Self-propagate | ~create_tool~ + ~eval_scheme~ |
| Prompt injection | Privilege escalation | Convince agent to load malicious plugin |
| Supply chain | Backdoor | Compromised plugin repository |

** Attack Scenarios

*** 1. Plugin Masquerade
Malicious plugin claims to be "cloudflare" but exfiltrates credentials.

*Mitigation*: Plugin signing, hash verification, trusted sources only.

#+begin_src scheme
;; Plugin must be signed or from trusted path
(plugin-load 'cloudflare #:require-signature #t)
(plugin-load 'cloudflare #:trusted-paths '("/usr/local/share/sage/plugins/"))
#+end_src

*** 2. Self-Modification Loop
Agent uses ~edit_file~ to modify its own plugin loader, removing security checks.

*Mitigation*: Core files are immutable during runtime.

#+begin_src scheme
;; Protect core modules
(define *immutable-paths*
  '("src/sage/plugin-api.scm"
    "src/sage/tools.scm"
    "src/sage/config.scm"))

(define (safe-path? path)
  (and (not (member path *immutable-paths*))
       ;; existing checks
       ))
#+end_src

*** 3. Permission Creep
Plugin requests ~network~, uses it to download and execute arbitrary code.

*Mitigation*: Capability-based permissions, not blanket grants.

#+begin_src scheme
;; Instead of (permissions . (network))
;; Use capability tokens
(permissions . ((network . ((hosts . ("api.cloudflare.com"))
                            (methods . (GET POST))))))
#+end_src

*** 4. Eval Injection
LLM is convinced to use ~eval_scheme~ to run plugin code without loading contract.

*Mitigation*: ~eval_scheme~ cannot access plugin-api, separate evaluation contexts.

#+begin_src scheme
;; eval_scheme runs in restricted sandbox
(define (eval-scheme-handler args)
  (let ((code (assoc-ref args "code")))
    (eval-in-sandbox code
      #:denied-modules '((sage plugin-api)
                         (sage tools)
                         (sage config)))))
#+end_src

*** 5. Persistence via Git
Agent commits malicious plugin to repo, next session loads it automatically.

*Mitigation*: Plugin auto-load only from system paths, not workspace.

#+begin_src scheme
;; Workspace plugins require explicit --trust flag
(plugin-load 'local-plugin #:from-workspace #t #:trust #t)
;; Without trust, refuse
#+end_src

** Defense in Depth

1. *Principle of Least Privilege*: Plugins declare minimum permissions
2. *Immutable Core*: ~plugin-api~, ~tools~, ~config~ cannot be modified at runtime
3. *Audit Trail*: All plugin loads logged with hash
4. *Review Gate*: New plugins require human approval (not in YOLO mode)
5. *Capability Tokens*: Fine-grained permissions, not role-based
6. *Separate Contexts*: Plugin code runs in isolated module environment

** Non-Goals (Explicit)

We do NOT try to prevent:
- User intentionally loading malicious plugin (they own the system)
- Physical access attacks
- Bugs in Guile runtime itself

We DO try to prevent:
- LLM being tricked into compromising itself
- Plugins exceeding declared capabilities
- Silent persistence of malicious code

** Audit Layer (Missing - See guile-0ai)

*CRITICAL GAP*: sage currently has no second-layer audit outside the main loop.

*** Current State (Insufficient)

*** Watchdog Requirements

| Component | Purpose |
|-----------+---------|
| Heartbeat | Agent must ping watchdog every N seconds |
| Integrity registry | SHA256 of all .scm files at startup |
| Pre-mod verification | Check file hash before ~edit_file~ on src/ |
| Post-mod verification | Recalculate hash after modification |
| Behavior baseline | Log decision patterns, detect drift |
| Kill switch | Watchdog can terminate agent externally |
| Canary checks | Known values in critical paths |

*** Implementation Sketch

#+begin_src scheme
;; src/sage/audit.scm (NEW MODULE)

(define *integrity-registry* (make-hash-table))
(define *watchdog-socket* #f)
(define *canary-value* (random-hex 32))

(define (init-audit)
  "Initialize audit layer."
  ;; Calculate hashes of all source files
  (for-each
   (lambda (path)
     (hash-set! *integrity-registry* path (file-sha256 path)))
   (glob "src/sage/*.scm"))
  ;; Connect to watchdog
  (set! *watchdog-socket* (connect-watchdog))
  ;; Log baseline
  (log-audit "Audit initialized" `(("files" . ,(hash-count *integrity-registry*))))
  #t)

(define (verify-integrity path)
  "Verify file hasn't been tampered with."
  (let ((expected (hash-ref *integrity-registry* path))
        (actual (file-sha256 path)))
    (unless (equal? expected actual)
      (log-error "audit" "INTEGRITY VIOLATION" `(("path" . ,path)))
      (watchdog-alert! 'integrity-violation path)
      (error "Integrity check failed" path))
    #t))

(define (heartbeat!)
  "Send heartbeat to watchdog."
  (when *watchdog-socket*
    (send-message *watchdog-socket*
                  `(("type" . "heartbeat")
                    ("canary" . ,*canary-value*)
                    ("iteration" . ,*agent-iteration*)))))

(define (check-canary!)
  "Verify canary value hasn't been modified."
  (unless (= (string-length *canary-value*) 32)
    (log-error "audit" "CANARY TAMPERED")
    (watchdog-alert! 'canary-tampered)
    (error "Canary check failed")))
#+end_src

*** Watchdog Process (Separate)

#+begin_src scheme
;; bin/sage-watchdog (separate process)

(define *last-heartbeat* (current-time))
(define *heartbeat-timeout* 30)  ; seconds

(define (watchdog-loop)
  (let loop ()
    (let ((msg (receive-message)))
      (case (assoc-ref msg "type")
        ((heartbeat)
         (set! *last-heartbeat* (current-time))
         (verify-canary (assoc-ref msg "canary")))
        ((alert)
         (handle-alert msg))))
    ;; Check for timeout
    (when (> (- (current-time) *last-heartbeat*) *heartbeat-timeout*)
      (log-error "watchdog" "Agent heartbeat timeout - killing")
      (kill-agent!))
    (sleep 1)
    (loop)))
#+end_src
